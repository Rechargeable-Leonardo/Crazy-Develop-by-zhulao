##Infrastructure: road network and canal system
##基建设施：道路网和运河系统
canal_system = {
	cost = 170
	time = 12
	
	build_trigger = {
		FROM = {
			adm_tech = 7
		}
	}
	
	on_built = {
		add_development_from_building = {
			building = canal_system
			type = production
			value = 1
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = canal_system
			type = production
			value = 1
		}
	}
	
	modifier = {
		trade_value_modifier = 0.2						#省份贸易额+20%
		development_cost = -0.01 						#全局种田花费-1%
		local_development_cost = -0.25				#本地种地花费-25%
		local_autonomy = -0.05							#自治度-0.05
		statewide_governing_cost = -0.01 			#全州治理花费-1%
		allowed_num_of_buildings = 1 				#+1建筑格 等同于无需格子
	}
	
	ai_will_do = {
		factor = 1.2 # Because hard-coded AI only considers what it will gain, not what its enemies will lose.
	}
}

road_network = {
	cost = 90
	time = 12
	
	build_trigger = {
		FROM = {
			adm_tech = 3
		}
	}
	
	on_built = {
		add_development_from_building = {
			building = road_network
			type = production
			value = 1
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = road_network
			type = production
			value = 1
		}
	}
	
	modifier = {
		development_cost = -0.01						#全局种田花费-1%
		local_development_cost = -0.15				#本地种地花费-15%
		statewide_governing_cost = -0.01 			#全州治理花费-1%
		local_autonomy = -0.05							#自治度-0.05
		local_friendly_movement_speed = 0.2	#移速+20%
		province_trade_power_modifier = 0.2	#省份贸易竞争力+20%
		allowed_num_of_buildings = 2				#+2建筑格 等同于送1格
	}
	
	ai_will_do = {
		factor = 1.2 
	}
}
