##Roads stuff
canal_system = {
	cost = 170
	time = 12
	
	build_trigger = {
		FROM = {
			adm_tech = 7
		}
	}
	
	on_built = {
		add_development_from_building = {
			building = canal_system
			type = production
			value = 1
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = canal_system
			type = production
			value = 1
		}
	}
	
	modifier = {
		trade_value_modifier = 0.1
		local_development_cost = -0.05
		allowed_num_of_buildings = 1
	}
	
	ai_will_do = {
		factor = 1.2 # Because hard-coded AI only considers what it will gain, not what its enemies will lose.
	}
}
