agri_farmland = {	#农田：规模化农场
	cost = 85
	time = 12

	bonus_manufactory = { 	# Will apply an extra modifier if these goods
		grain				#谷物、鱼、牲畜、酒类时获得额外人力加成
		fish
		livestock
		wine
	}

	build_trigger = {
		has_terrain = farmlands	#农田地形可用
		FROM = { adm_tech = 4 }	#4级adm
	}
	
	on_built = {
		add_development_from_building = {
			building = agri_farmland
			type = tax
			value = 2
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = agri_farmland
			type = tax
			value = 2
		}
	}
	
	modifier = {
		local_friendly_movement_speed = 0.1	#移动速度补正
		local_manpower = 0.05 				#士兵之家三十分之一的人力
		local_development_cost = -0.45		#种田花费从-5%提升至-50%
		allowed_num_of_buildings = 1		#增加1个格子，相当于无需格子
	}
	bonus_modifier = {
		local_manpower = 0.05		#士兵之家三十分之一的人力
		trade_goods_size = 0.05 	#0.05商品产出
	}
	
	ai_will_do = {
		factor = 1				#反正默认是1，我也懒得改辽
	}
}
