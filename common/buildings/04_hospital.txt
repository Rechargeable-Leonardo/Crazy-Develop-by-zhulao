sick_house = {	#1级医院：病患小屋
	cost = 25
	time = 1

	modifier = {
		local_development_cost = -0.05			#本地种地花费-5%
		local_monthly_devastation = -0.01		#本地荒废-0.01
	}
	on_built = {
		add_development_from_building = {
			building = sick_house
			type = manpower
			value = 1
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = sick_house
			type = manpower
			value = 1
		}
	}
	on_obsolete = {
	}

	ai_will_do = {
		factor = 1.5
	}
}
