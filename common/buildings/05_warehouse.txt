grain_warehouse = {		#粮仓
	cost = 150
	time = 24

	bonus_manufactory = { 	# Will apply an extra modifier if these goods
		grain				#谷物、鱼、牲畜、酒类时获得额外加成
		fish
		livestock
		wine
	}

	modifier = {
		local_development_cost = -0.10 			#本地种田花费-10%
		local_manpower = 0.01 					#人力：固定值10个
		local_manpower_modifier = 0.05			#人力：5%
		local_monthly_devastation = -0.01		#本地荒废-0.01
		local_unrest = -0.5						#本地叛乱-0.5
		supply_limit = 3						#补给上限：固定值3
		supply_limit_modifier = 0.1				#补给上限：10%
	}
	bonus_modifier = {
		local_monthly_devastation = -0.01		#本地荒废-0.01
		local_unrest = -0.5						#本地叛乱-0.5
		supply_limit = 3						#补给上限：固定值3
		supply_limit_modifier = 0.1				#补给上限：10%
	}

	on_built = {
		add_development_from_building = {
			building = grain_warehouse
			type = manpower
			value = 1
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = grain_warehouse
			type = manpower
			value = 1
		}
	}
	on_obsolete = {
	}

	ai_will_do = {
		factor = 1.5 # Because hard-coded AI only considers what it will gain, not what its enemies will lose.
	}
}
