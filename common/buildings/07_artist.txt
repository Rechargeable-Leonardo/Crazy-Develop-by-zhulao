art_museum = {		#1级艺术存储建筑-博物馆
	cost = 150
	time = 12
	
	build_trigger = {
		development = 5
	}
	
	on_built = {
		add_development_from_building = {
			building = art_museum
			type = tax
			value = 1
		}
		hidden_effect = {
			change_variable = { which = loacl_artwork_slot value = 1 }
			change_variable = { which = loacl_art_strength value = 1 }
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = art_museum
			type = tax
			value = 1
		}
		hidden_effect = {
			subtract_variable = { which = loacl_artwork_slot value = 1 }
			subtract_variable = { which = loacl_art_strength value = 1 }
		}
	}
	
	modifier = {
		local_missionary_strength = 0.001		#本地传教力量0.1%
		local_culture_conversion_cost = -0.1	#本地转文化花费-10%
		local_institution_spread = 0.01			#本地思潮传播+1%
		local_unrest = -0.5								#本地叛乱-0.5
	}
	
	ai_will_do = {
		factor = 1.2 # Because hard-coded AI only considers what it will gain, not what its enemies will lose.
	}
}
